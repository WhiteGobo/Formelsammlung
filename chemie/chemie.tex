\newcommand{\CommonElementTextFormat}[5]
{

	\centering
	{\textbf{#1} \hfill #2}%
	\linebreak \linebreak
	{\textbf{#3}}%
	\linebreak
	{{#4}} %  \\%
	\linebreak
	{{#5} \hfill { }}

}
